*******************************************************************************
This query shows all table-level granted privileges for a user named 'MYUSER':
*******************************************************************************
select
   owner,GRANTEE,
   table_name,
   select_priv,
   insert_priv,
   delete_priv,
   update_priv,
   references_priv,
   alter_priv,
   index_priv
from
   table_privileges
where
   grantee = 'A81FPX'
order by
   owner,
   table_name;
*******************************************************************************
This query shows all role privileges for a user:
*******************************************************************************
select distinct
   owner,
   table_name,
   privilege
from
   dba_role_privs rp,
   role_tab_privs rtp
where
   rp.granted_role = rtp.role
and
   rp.grantee = 'A81FPX'
order by
   owner,
   table_name;
