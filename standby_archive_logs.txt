#!/bin/bash

export FILE_DIR=/home/oracle/Script

#Delete if file exists

[ -f $FILE_DIR/Execution_INSISDB.log ] &&  rm $FILE_DIR/Execution_INSISDB.log
[ -f $FILE_DIR/Execution_REPOSDR.log ] &&  rm $FILE_DIR/Execution_REPOSDR.log

# Set the Environment Variables

export ORACLE_SID=INSISDB1
export ORACLE_HOME=/u01/app/oracle/rdbms19c
export LD_LIBRARY_PATH=/u01/app/oracle/rdbms19c/lib
export PATH=/u01/app/oracle/rdbms19c:$PATH

# Login to rman for the INSISDB1 FRA clean-up

PRE="set pagesize 0 \n set feedback off \n";
SS="$ORACLE_HOME/bin/sqlplus -L -S / as sysdba"
ROLE=$(echo -e "$PRE select database_role from v\$database;" | $SS)
DATABASE=$(echo -e "$PRE select db_unique_name from v\$database;" | $SS)

echo "************Database Information*************" >> $FILE_DIR/Execution_INSISDB.log
echo "Database Name is: $DATABASE" >> $FILE_DIR/Execution_INSISDB.log
echo "Database Role is: $ROLE" >> $FILE_DIR/Execution_INSISDB.log
printf "\n\n*****************************************************************************************************************************\n" >> $FILE_DIR/Execution_INSISDB.log

[[ "$ROLE" != "PHYSICAL STANDBY" ]] && { echo "ERROR: database not a physical standby" >> $FILE_DIR/Execution_INSISDB.log; exit 1; }

THREADS=$(echo -e "$PRE select distinct thread# from v\$archived_log;" | $SS)

for THREAD in $THREADS; do
  #MAX_APPLIED=$(echo -e "$PRE select max(sequence#) from v\$archived_log where applied='YES' and thread#=$THREAD and COMPLETION_TIME < SYSDATE-1/2;" | $SS)
  MAX_APPLIED=$(echo -e "$PRE select max(sequence#) from v\$archived_log where applied='YES' and thread#=$THREAD;" | $SS)
  MAX_APPLIED=`expr $MAX_APPLIED - 20`;
  echo "Deleting the log for Database $DATABASE Thread $THREAD Until Sequence $MAX_APPLIED" >> $FILE_DIR/Execution_INSISDB.log
  echo "*****************************************************************************************************************************" >> $FILE_DIR/Execution_INSISDB.log
  echo "delete noprompt archivelog until sequence $MAX_APPLIED thread $THREAD;"|$ORACLE_HOME/bin/rman target / >> $FILE_DIR/Execution_INSISDB.log
  echo "*****************************************************************************************************************************" >> $FILE_DIR/Execution_INSISDB.log
  echo "Deleted the log for Database $DATABASE Thread $THREAD that satisfied the condition 2 days old and Archived log applied on DR"  >> $FILE_DIR/Execution_INSISDB.log
  echo "*****************************************************************************************************************************" >> $FILE_DIR/Execution_INSISDB.log
done

# Login to sqlplus for the INSISDB ARCHIVED usage

PCT_USAGE_INSISDB=$(${ORACLE_HOME}/bin/sqlplus -S "/ as sysdba" <<EOF
set pages 0 feedback off ;
select round(PERCENT_SPACE_USED) from  V\$FLASH_RECOVERY_AREA_USAGE where FILE_TYPE='ARCHIVED LOG';
exit;
EOF
)

PCT_USAGE_INSISDB=${PCT_USAGE_INSISDB%%.*}

# If Usage is more then 79%% for INSISDB then sent email

if (( ${PCT_USAGE_INSISDB} >  80 ));then
echo "*****************************Archive Log Usage*******************************" >> $FILE_DIR/Execution_INSISDB.log
echo "Archive Log usage percentage is: $PCT_USAGE_INSISDB" >> $FILE_DIR/Execution_INSISDB.log
echo "*****************************************************************************" >> $FILE_DIR/Execution_INSISDB.log
echo -e "Hi Team,\n\nArchive log usage is High. DR node : $HOSTNAME \n\nPlease find the log in attachment and look into it immediately. \n\nThanks & Regards,\nInfosys Ettsak Infra Team" | mailx -s "ARCHIVE LOG Area usage for Database $DATABASE" -a $FILE_DIR/Execution_INSISDB.log -r Infosys-Ettsak-Infra-Team@lansforsakringar.se neeraj.gupta@lansforsakringar.se Infosys-Ettsak-Infra-Team@lansforsakringar.se
fi

echo -e "Hi Team,\n\nArchive log successfully cleaned in DR node : $HOSTNAME \n\nPlease find the log in attachment. \n\nThanks & Regards,\nInfosys Ettsak Infra Team" | mailx -s "ARCHIVE LOG cleanup $DATABASE" -a $FILE_DIR/Execution_INSISDB.log neeraj.gupta@lansforsakringar.se Konsult.Karthikeyan.Saravanan@lansforsakringar.se


# Login to rman for the REPOSDR FRA clear-up

export ORACLE_SID=REPOSDR1
PRE="set pagesize 0 \n set feedback off \n";
SS="$ORACLE_HOME/bin/sqlplus -L -S / as sysdba"
ROLE=$(echo -e "$PRE select database_role from v\$database;" | $SS)
DATABASE=$(echo -e "$PRE select db_unique_name from v\$database;" | $SS)

echo "************Database Information*************" >> $FILE_DIR/Execution_REPOSDR.log
echo "Database Name is: $DATABASE" >> $FILE_DIR/Execution_REPOSDR.log
echo "Database Role is: $ROLE" >> $FILE_DIR/Execution_REPOSDR.log
printf "\n\n*****************************************************************************************************************************\n" >> $FILE_DIR/Execution_REPOSDR.log

[[ "$ROLE" != "PHYSICAL STANDBY" ]] && { echo "ERROR: database not a physical standby" >> $FILE_DIR/Execution_REPOSDB.log; exit 1; }

THREADS=$(echo -e "$PRE select distinct thread# from v\$archived_log;" | $SS)

for THREAD in $THREADS; do
  MAX_APPLIED=$(echo -e "$PRE select max(sequence#) from v\$archived_log where applied='YES' and thread#=$THREAD and COMPLETION_TIME < SYSDATE-1/2;" | $SS)
  echo "Deleting the log for Database $DATABASE Thread $THREAD Until Sequence $MAX_APPLIED" >> $FILE_DIR/Execution_REPOSDR.log
  echo "*****************************************************************************************************************************" >> $FILE_DIR/Execution_REPOSDR.log
  echo "delete noprompt archivelog until sequence $MAX_APPLIED thread $THREAD;"|$ORACLE_HOME/bin/rman target / >> $FILE_DIR/Execution_REPOSDR.log
  echo "*****************************************************************************************************************************" >> $FILE_DIR/Execution_REPOSDR.log
  echo "Deleted the log for Database $DATABASE Thread $THREAD that satisfied the condition 2 days old and Archived log applied on DR"  >> $FILE_DIR/Execution_REPOSDR.log
  echo "*****************************************************************************************************************************" >> $FILE_DIR/Execution_REPOSDR.log
done


# Login to sqlplus for the REPOSDR. ARCHIVED usage

PCT_USAGE_REPOSDR=$(${ORACLE_HOME}/bin/sqlplus -S "/ as sysdba" <<EOF
set pages 0 feedback off ;
select round(PERCENT_SPACE_USED) from  V\$FLASH_RECOVERY_AREA_USAGE where FILE_TYPE='ARCHIVED LOG';
exit;
EOF
)

PCT_USAGE_REPOSDR=${PCT_USAGE_REPOSDR%%,*}

# If Usage is more then 79%% for REPOSDR then sent email

if (( ${PCT_USAGE_REPOSDR} >  79 ));then
echo "*****************************Archive Log Usage*******************************" >> $FILE_DIR/Execution_REPOSDR.log
echo "Archive Log usage percentage is: $PCT_USAGE_REPOSDR" >> $FILE_DIR/Execution_REPOSDR.log
echo "*****************************************************************************" >> $FILE_DIR/Execution_REPOSDR.log
echo -e "Hi Team,\n\nArchive log usage is High. DR node : $HOSTNAME \n\nPlease find the log in attachment and look into it immediately. \n\nThanks & Regards,\nInfosys Ettsak Infra Team" | mailx -s "ARCHIVE LOG Area usage for Database $DATABASE" -a $FILE_DIR/Execution_REPOSDR.log neeraj.gupta@lansforsakringar.se
fi

echo -e "Hi Team,\n\nArchive log successfully cleaned in DR node : $HOSTNAME \n\nPlease find the log in attachment. \n\nThanks & Regards,\nInfosys Ettsak Infra Team" | mailx -s "ARCHIVE LOG cleanup $DATABASE" -a $FILE_DIR/Execution_REPOSDR.log neeraj.gupta@lansforsakringar.se
