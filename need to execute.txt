5. Primary Database: Clear Blocking Scheduler Jobs

    -- Step 1: Check and stop the job queue
    
  SHOW PARAMETER job_queue_processes;
  ALTER SYSTEM SET job_queue_processes = 0 SCOPE=BOTH;
      => Blocks new jobs from starting, resume after switchover!

    -- Step 2: List All Active (Non-SYS) Scheduler Jobs
    COL owner FOR a20;
    COL job_name FOR a30;
    COL start_date FOR a50;
    COL end_date FOR a20;
    SELECT owner, job_name, start_date, end_date, enabled
      FROM dba_scheduler_jobs
     WHERE enabled = 'TRUE' AND owner <> 'SYS';

    -- Step 3: Generate Disable Statements
    SELECT 'EXECUTE DBMS_SCHEDULER.DISABLE(''' || owner || '.' || job_name || ''');'
      FROM dba_scheduler_jobs
     WHERE enabled = 'TRUE' AND owner <> 'SYS';

    -- Step 4: Manually Stop Jobs if Needed (example)
    BEGIN
      DBMS_SCHEDULER.STOP_JOB('INSIS_CUST.RUN_CHAIN$UDP_CHAI3578');
    END;
    /

    *Note: Always record which jobs you disable or stop, and return them to their original states post-switchover.
